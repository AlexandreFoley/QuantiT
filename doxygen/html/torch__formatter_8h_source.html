29template <>
30struct fmt::formatter<torch::Tensor>
31{
32 uint linelenght = 80;
33
34 constexpr auto parse(format_parse_context& ctx)
35 {
36 auto it = ctx.begin(), end = ctx.end();
37 if (it and *it != '}')
38 {
39 ++it;
40 auto first_pos = it;
41 while (it != end && *it != '}')
42 {
43 ++it;
44 }
45 auto last_pos = it - 1;
46 uint width = 80;
47 if (last_pos > first_pos)
48 {
49 auto code = std::from_chars(first_pos, last_pos, width);
50 if (code.ptr != last_pos || std::errc::invalid_argument == code.ec)
51 throw format_error("invalid format, only a single integer supported for torch::Tensor");
52 else
53 {
54 linelenght = width;
55 }
56 }
57 }
58 if (it and *it != '}')
59 throw format_error("invalid format,closing brace missing");
60
61 // Return an iterator past the end of the parsed range:
62 return it;
63 }
64
65 template <typename FormatContext>
66 auto format(const torch::Tensor &tens, FormatContext &ctx) const
67 {
68 std::stringstream strstr;
69 at::print(strstr, tens, linelenght);
70 std::string tens_string(strstr.str());
71 auto end = tens_string.length();
72 // find occurences of {} and replace with ()
73 // done because fmt use {} to mark format section which can lead to weird interactions...
74 auto it = tens_string.find("{", 0);
75 while (it < end)
76 {
77 tens_string[it] = '(';
78 ++it;
79 it = tens_string.find("{", it);
80 }
81 it = tens_string.find("}", 0);
82 while (it < end)
83 {
84 tens_string[it] = ')';
85 ++it;
86 it = tens_string.find("}", it);
87 }
88
89 return format_to(ctx.out(), tens_string);
90 }
91};
