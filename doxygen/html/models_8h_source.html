44qtt_TEST_CASE("Heisenberg")
45{
46 int J(1);
47 // qtt_REQUIRE(J.isIntegral(false) );
48 auto Heis = details::Heisenberg_impl(torch::tensor(J),3);
49 qtt_CHECK(Heis[0].sizes().size() == 4);
50 auto size_left_edge = std::vector<int64_t>{1l,2l,5l,2l};
51 qtt_CHECK(Heis[0].sizes().vec() == size_left_edge);
52 qtt_CHECK(Heis[1].sizes().size() == 4);
53 auto size_middle = std::vector<int64_t>{5l,2l,5l,2l};
54 qtt_CHECK(Heis[1].sizes().vec() == size_middle);
55 qtt_CHECK(Heis[2].sizes().size() == 4);
56 auto size_right_edge = std::vector<int64_t>{5l,2l,1l,2l};
57 qtt_CHECK(Heis[2].sizes().vec() == size_right_edge);
58
59 qtt_REQUIRE(!Heis[0].is_floating_point());
60
61
62 auto test_Heis = torch::zeros({8,8},torch::kInt8);
63 {
64 auto acc = test_Heis.accessor<int8_t,2>();
65 acc[0][0] = acc[2][1] = acc[1][2] = 2;
66 acc[4][2] = acc[2][4] = acc[5][3] = acc[3][5] = 2;
67 acc[7][7] = acc[6][5] = acc[5][6] = 2;
68 acc[2][2] = acc[5][5] = -2;
69 }
70
71 auto cont_Heis = torch::tensordot(Heis[0],Heis[1],{2},{0}).permute({0,1,3,4,2,5}).reshape({1,4,5,4});
72 cont_Heis = torch::tensordot(cont_Heis,Heis[2],{2},{0}).permute({0,1,3,4,2,5}).reshape({8,8});
73 qtt_CHECK(torch::equal(cont_Heis,test_Heis));
74}
