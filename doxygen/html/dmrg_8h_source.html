115 qtt_TEST_CASE("2x2 eigen value problem")
116 {
117  // setup: a random answer from which we construct a matrix
118  auto angle = torch::rand({}, torch::kFloat64) * 2 * M_PI;
119  auto E0 = torch::rand({}, torch::kFloat64) * -1;
120  auto E1 = torch::rand({}, torch::kFloat64);
121  auto psi00 = torch::zeros({}, torch::kFloat64);
122  auto psi01 = torch::zeros({}, torch::kFloat64);
123  psi00 = cos(angle);
124  psi01 = sin(angle);
125  auto psi10 = torch::zeros({}, torch::kFloat64);
126  auto psi11 = torch::zeros({}, torch::kFloat64);
127  psi10 = sin(angle);
128  psi11 = -cos(angle);
129  auto a0 = psi00.pow(2) * E0 + psi10.pow(2) * E1; // matrix element
130  auto a1 = psi01.pow(2) * E0 + psi11.pow(2) * E1; // matrix element
131  auto b = psi01 * psi00 * E0 + psi11 * psi10 * E1; // matrix element
132 
133  auto [T_E0, T_psi00, T_psi01] = details::eig2x2Mat(a0, a1, b);
134  // for the phase gauge freedom.
135  bool state_check = (torch::allclose(psi00, T_psi00) and torch::allclose(psi01, T_psi01)) or
136  (torch::allclose(-1 * psi00, T_psi00) and torch::allclose(-1 * psi01, T_psi01));
137  qtt_CHECK(state_check);
138  qtt_CHECK(torch::allclose(T_E0, E0));
139 }

152  acc[1][1][0][1] = acc[2][1][1][1] = -1;
153  //Identities
154  acc[0][0][0][0] = acc[0][1][0][1] = 1;
155  acc[2][0][2][0] = acc[2][1][2][1] = 1;

157  using cval = quantity<conserved::Z>;
158  auto bl_ising = btensor({{{3,cval(0)}},{{1,cval(1)},{1,cval(-1)}},{{3,cval(0)}},{{1,cval(-1)},{1,cval(1)}}},cval(0));
159  bl_ising = from_basic_tensor_like(bl_ising,l_ising);
160  bMPO ising(2, bl_ising);
161  ising[0] = ising[0].basic_create_view({2, -1,-1,-1},true);//preserve the rank when creating the view
162  ising[1] = ising[1].basic_create_view({-1, -1, 0,-1},true);//preserve the rank when create the view.

172  auto norm = contract(rstate,rstate);
173  qtt_CHECK(allclose(norm, quantit::ones({},cval(0))));
174  // fmt::print("Norm2 \n{}\n",contract(rstate,rstate));

176  // fmt::print("{}\n\n",squeeze(rstate[1]));

178  // fmt::print("{}\n\n",two_s_state);

181  qtt_CHECK(allclose(avg,ones_like(avg) ));
182 
183  auto H_av = contract(rstate, rstate, ising);
