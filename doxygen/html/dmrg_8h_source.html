115qtt_TEST_CASE("2x2 eigen value problem")
116{
117 // setup: a random answer from which we construct a matrix
118 auto angle = torch::rand({}, torch::kFloat64) * 2 * M_PI;
119 auto E0 = torch::rand({}, torch::kFloat64) * -1;
120 auto E1 = torch::rand({}, torch::kFloat64);
121 auto psi00 = torch::zeros({}, torch::kFloat64);
122 auto psi01 = torch::zeros({}, torch::kFloat64);
123 psi00 = cos(angle);
124 psi01 = sin(angle);
125 auto psi10 = torch::zeros({}, torch::kFloat64);
126 auto psi11 = torch::zeros({}, torch::kFloat64);
127 psi10 = sin(angle);
128 psi11 = -cos(angle);
129 auto a0 = psi00.pow(2) * E0 + psi10.pow(2) * E1; // matrix element
130 auto a1 = psi01.pow(2) * E0 + psi11.pow(2) * E1; // matrix element
131 auto b = psi01 * psi00 * E0 + psi11 * psi10 * E1; // matrix element
132
133 auto [T_E0, T_psi00, T_psi01] = details::eig2x2Mat(a0, a1, b);
134 // for the phase gauge freedom.
135 bool state_check = (torch::allclose(psi00, T_psi00) and torch::allclose(psi01, T_psi01)) or
136 (torch::allclose(-1 * psi00, T_psi00) and torch::allclose(-1 * psi01, T_psi01));
137 qtt_CHECK(state_check);
138 qtt_CHECK(torch::allclose(T_E0, E0));
139}
