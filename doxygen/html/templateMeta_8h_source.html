120template <class Tuple1, class F>
121constexpr decltype(auto) for_each(Tuple1 &&T1, F &&f)
122{
123 return std::apply(
124 [&](auto &&... t1) {
125 if constexpr (std::is_same_v<decltype(f(std::get<0>(T1))), void>)
126 { // we return void..
127 (f(t1), ...); // apply the function to each arguments
128 }
129 else
130 {
131 return std::tuple(f(t1)...); // apply the function to each agruments and form
132 // a tuple with the result.
133 }
134 },
135 T1);
136}
148template <class Tuple1, class Tuple2, class F>
149constexpr decltype(auto) for_each2(Tuple1 &&T1, Tuple2 &&T2, F &&f)
150{
151 return std::apply(
152 [&](auto &&... t1) { // std::apply require gcc7 or more recent
153 return std::apply(
154 [&T1, &T2, &f, &t1...](auto &&... t2) {
155 // need explicit capture list here for clang8 and older.
156 if constexpr (std::is_same_v<decltype(f(std::get<0>(T1), std::get<0>(T2))), void>)
157 { // we return void..
158 (f(t1, t2), ...); // apply the function to each pair of arguments
159 }
160 else
161 {
162 return std::tuple(f(t1, t2)...); // apply the function to each pair and form
163 // a tuple with the result.
164 }
165 },
166 T2);
167 },
168 T1);
