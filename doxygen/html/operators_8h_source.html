64qtt_TEST_CASE("half spin fermions")
65{
66 auto [c_up, c_dn, F, id] = fermions();
67
68 auto cd_up = c_up.conj().t();
69 auto cd_dn = c_dn.conj().t();
70 // The specific content of those matrices isn't all that important, so long as they have the following properties
71 // anticommutation
72 qtt_CHECK(torch::equal(torch::matmul(cd_up, c_up), id - torch::matmul(c_up, cd_up)));
73 qtt_CHECK(torch::equal(torch::matmul(cd_dn, c_dn), id - torch::matmul(c_dn, cd_dn)));
74
75 qtt_CHECK(torch::equal(torch::matmul(cd_up, c_dn), -1 * torch::matmul(c_dn, cd_up)));
76 qtt_CHECK(torch::equal(torch::matmul(cd_dn, c_up), -1 * torch::matmul(c_up, cd_dn)));
77
78 qtt_CHECK(torch::equal(torch::matmul(c_up, c_dn), -1 * torch::matmul(c_dn, c_up)));
79 qtt_CHECK(torch::equal(torch::matmul(c_up, c_up), -1 * torch::matmul(c_up, c_up)));
80 qtt_CHECK(torch::equal(torch::matmul(c_dn, c_dn), -1 * torch::matmul(c_dn, c_dn)));
81
82 qtt_CHECK(torch::equal(torch::matmul(c_up, F), -1 * torch::matmul(F, c_up)));
83 qtt_CHECK(torch::equal(torch::matmul(c_dn, F), -1 * torch::matmul(F, c_dn)));
84 // F^2 = 1
85 qtt_CHECK(torch::equal(torch::matmul(F, F), id));
86 // identity
87 qtt_CHECK(torch::equal(torch::matmul(id, id), id));
88 qtt_CHECK(torch::equal(torch::matmul(id, c_up), c_up));
89 qtt_CHECK(torch::equal(torch::matmul(id, F), F));
90 qtt_CHECK(torch::equal(torch::matmul(id, c_dn), c_dn));
91}

103qtt_TEST_CASE("conserved quantities fermions")
104{
105 using cvals = quantity<conserved::Z>;
106 auto shape = btensor({{{1, cvals(0)}, {2, cvals(1)}, {1, cvals(2)}}}, cvals(0));
107 auto op_shape = shape_from(shape, shape.conj());
108
109 auto [c_up, c_dn, F, id] = fermions(op_shape);
110 qtt_CHECK(c_up.selection_rule->get() == cvals(-1));
111 qtt_CHECK(c_dn.selection_rule->get() == cvals(-1));
112
113 qtt_SUBCASE("insertion into MPO tensor")
114 {
115 auto U = full({}, cvals(0), 6);
116 auto mu = full({}, cvals(0), 3);
117 auto t = full({}, cvals(0), 1);
118 auto c_dag_up = c_up.conj().permute({1, 0});
119 auto c_dag_dn = c_dn.conj().permute({1, 0});
120 auto leftbond = btensor(
121 {{{1, cvals(0)}, {1, cvals(1)}, {1, cvals(1)}, {1, cvals(-1)}, {1, cvals(-1)}, {1, cvals(0)}}}, cvals(0));
122 auto T = shape_from(leftbond, shape, leftbond.conj(), shape.conj());
123
124 auto n_up = c_dag_up.bmm(c_up);
125 auto n_dn = c_dag_dn.bmm(c_dn);
126
127 auto H_l = -mu * (n_up + n_dn) + U * (n_up.bmm(n_dn));
128
129 T.basic_index_put_({0, -1, 0, -1}, id);
130 T.basic_index_put_({1, -1, 0, -1}, c_up);
131 T.basic_index_put_({2, -1, 0, -1}, c_dn);
132 T.basic_index_put_({ 3, -1, 0, -1 }, c_dag_up);
133 T.basic_index_put_({ 4, -1, 0, -1 }, c_dag_dn);
134 T.basic_index_put_({ 5, -1, 0, -1 }, H_l);
135 T.basic_index_put_({ 5, -1, 1, -1 }, t * F.bmm(c_dag_up));
136 T.basic_index_put_({ 5, -1, 2, -1 }, t * F.bmm(c_dag_dn));
137 T.basic_index_put_({ 5, -1, 3, -1 }, t * c_up.bmm( F));
138 T.basic_index_put_({ 5, -1, 4, -1 }, t * c_dn.bmm( F));
139 T.basic_index_put_({ 5, -1, 5, -1 }, id);
140 }
141}
