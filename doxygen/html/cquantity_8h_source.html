315 qtt_TEST_CASE("composite conserved")
316 {
317  using namespace quantit;
318  using namespace conserved;
319  any_quantity A(C<2>(0), Z(3)); // order matters.
320  any_quantity B(C<2>(1), Z(-1));
321  using ccgroup = quantity<C<2>, Z>;
322  auto EFF = any_quantity(ccgroup(0, 0));
323  qtt_CHECK_NOTHROW(auto a = any_quantity(ccgroup(0, 0)));
324  any_quantity A_copy(A);
325  any_quantity B_copy(B);
326  qtt_CHECK_NOTHROW(auto c = A + B);
327  any_quantity D(Z(0), C<2>(1)); // D has a different underlying type.
328  // the cast to void silences warnings about unused return values and
329  // comparison. We know, it's ok.
330  qtt_CHECK_THROWS_AS((void)(D == A),
331  const std::bad_cast &); // A and D have different derived
332  // type: they're not compatible
333  qtt_CHECK_THROWS_AS((void)(D != A),
334  const std::bad_cast &); // A and D have different derived
335  // type: they're not compatible
336  qtt_CHECK_THROWS_AS((void)(D * A),
337  const std::bad_cast &); // A and D have different derived
338  // type: they're not compatible
339  qtt_CHECK_THROWS_AS(D *= A,
340  const std::bad_cast &); // A and D have different derived
341  // type: they're not compatible
342  qtt_CHECK_THROWS_AS(D += A,
343  const std::bad_cast &); // A and D have different derived
344  // type: they're not compatible
345  qtt_CHECK_THROWS_AS((void)(D + A),
346  const std::bad_cast &); // A and D have different derived
347  // type: they're not compatible
348  any_quantity_ref A_ref(A);
349  any_quantity_cref A_cref(A);
350  any_quantity_cref B_cref(B);
351  any_quantity_ref B_ref(B);
352  // any_quantity_ref is a drop-in replacement for a reference to any_quantity
353  // any_quantity_cref is a drop-in-replacement for a reference to a constant
354  // any_quantity. Those two classes exists to facilitate manipulation of single
355  // any_quantity located within a special container for this polymorphic type. We
356  // make extensive use of the reference type within the tests specifically to
357  // verify the correctness of their implementation. We advise avoiding those
358  // two reference type whenever possible. const any_quantity& and any_quantity& are
359  // perfectly fine for most purpose.
360  qtt_CHECK_NOTHROW(A_copy = A_ref);
361  qtt_CHECK_NOTHROW(A_copy = A_cref);
362  // Checking that the ref type doesn't loose track of its target.
363  qtt_CHECK(A_cref == A_ref);
364  qtt_CHECK(A_ref == A_cref);
365  qtt_CHECK(A_ref == A);
366  qtt_CHECK_NOTHROW(A_ref *= B);
367  qtt_CHECK(A_ref == A);
368  qtt_CHECK(A_ref == A_cref);
369  qtt_CHECK_NOTHROW(A_ref += B);
370  qtt_CHECK(A_ref == A);
371  qtt_CHECK(A_ref == A_cref);
372  A_copy = A;
373  qtt_CHECK(B == B_copy); // commute_ act on the the calling object only.
374  qtt_CHECK(A_copy == A); // this is an abelian group.
375  qtt_SUBCASE("Cast ambiguity")
376  {
377  // In this subcase with test different combination of any_quantity,any_quantity_ref
378  // and any_quantity_cref to make sure all operation resolve correctly. Because
379  // of the different type and the ways in which they are equivalent, a
380  // mistake could lead to some operations failing.
381  qtt_CHECK_NOTHROW(auto _t = A * B);
382  qtt_CHECK_NOTHROW(auto _t = A_ref * B);
383  qtt_CHECK_NOTHROW(auto _t = A_cref * B);
384  qtt_CHECK_NOTHROW(auto _t = A * B_ref);
385  qtt_CHECK_NOTHROW(auto _t = A_ref * B_ref);
386  qtt_CHECK_NOTHROW(auto _t = A_cref * B_ref);
387  qtt_CHECK_NOTHROW(auto _t = A * B_cref);
388  qtt_CHECK_NOTHROW(auto _t = A_ref * B_cref);
389  qtt_CHECK_NOTHROW(auto _t = A_cref * B_cref);
390  qtt_CHECK_NOTHROW(auto _t = A * B_cref.inverse()); // this should use the
391  // operator*(any_quantity_cref&,any_quantity&&);
392  qtt_CHECK_NOTHROW(auto _t = A_ref * B.inverse()); // this should use the
393  // operator*(any_quantity_cref&,any_quantity&&);
394  qtt_CHECK_NOTHROW(auto _t = A_cref * B_ref.inverse()); // this should use the
395  // operator*(any_quantity_cref&,any_quantity&&);
396  //+ and * are completly equivalent.
397  qtt_CHECK_NOTHROW(auto _t = A + B);
398  qtt_CHECK_NOTHROW(auto _t = A_ref + B);
399  qtt_CHECK_NOTHROW(auto _t = A_cref + B);
400  qtt_CHECK_NOTHROW(auto _t = A + B_ref);
401  qtt_CHECK_NOTHROW(auto _t = A_ref + B_ref);
402  qtt_CHECK_NOTHROW(auto _t = A_cref + B_ref);
403  qtt_CHECK_NOTHROW(auto _t = A + B_cref);
404  qtt_CHECK_NOTHROW(auto _t = A_ref + B_cref);
405  qtt_CHECK_NOTHROW(auto _t = A_cref + B_cref);
406  qtt_CHECK_NOTHROW(auto _t = A + B_cref.inverse()); // this should use the
407  // operator+(any_quantity_cref&,any_quantity&&);
408  qtt_CHECK_NOTHROW(auto _t = A_ref + B.inverse()); // this should use the
409  // operator+(any_quantity_cref&,any_quantity&&);
410  qtt_CHECK_NOTHROW(auto _t = A_cref + B_ref.inverse()); // this should use the
411  // operator+(any_quantity_cref&,any_quantity&&);
412  }
413  auto C = any_quantity(B_cref);
414  // C == A_ref *B_cref;
415  // C == B_cref *A_cref.inverse();
416  // C == A_ref *B_cref *A_cref.inverse(); //A.commute(B) should be
417  // an optimisation of the formula on the right.
418  qtt_CHECK(C == A_ref * B_cref * A_cref.inverse()); // A.commute(B) should be an optimisation
419  // of the formula on the right.
420  C = A;
421  qtt_CHECK(C == B.inverse() * A * B); // A.commute_(B) should be an optimization
422  // of the formula on the right.
423 }
