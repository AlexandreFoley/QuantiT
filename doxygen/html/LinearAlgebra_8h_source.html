119qtt_TEST_CASE("Linear Algebra for Tensor network")
120{
121 torch::set_default_dtype(torch::scalarTypeToTypeMeta(torch::kFloat64)); //otherwise the type promotion always goes to floats when promoting a tensor
122 auto split = 2;
123 auto tensor_shape = std::vector<int64_t>{10,3,10,3};
124 auto matrix_shape = std::vector<int64_t>{30,30};
125 auto u_shape = std::vector<int64_t>{10,3,30};
126 auto A = torch::rand(tensor_shape);
127 auto ra = A.reshape(matrix_shape);
128 auto [u_o,d_o,v_o] = ra.svd();
129 auto [u,d,v] = svd(A,2);
130
131 qtt_CHECK(u_o.sizes() == std::vector<int64_t>(matrix_shape));
132 qtt_REQUIRE(u.sizes() == std::vector<int64_t>(u_shape));
133
134 qtt_REQUIRE_NOTHROW(auto ru_o = u_o.reshape(u_shape));
135 auto ru_o = u_o.reshape(u_shape);
136
137 qtt_CHECK(torch::allclose(ru_o, u));
138
139 auto [e_o,s_o] = torch::linalg::eigh(ra,"L");
140 auto rs_o = s_o.reshape(u_shape);
141 auto [e,s] = eigh(A,2);
142
143 qtt_REQUIRE(s.sizes() == std::vector<int64_t>(u_shape));
144 // fmt::print("s {}\n",s);
145 // fmt::print("s_o {}\n",rs_o);
146 qtt_CHECK(torch::allclose(rs_o, s));
147}
