57qtt_TEST_CASE("dimension manipulation tools")
58{
59 std::vector<int64_t> dim1({2,3,4});
60 std::vector<int64_t> dim2({5,6});
61 std::vector<int64_t> dim3({7,8});
62
63 qtt_CHECK(prod(dim1) == 24);
64 qtt_CHECK(prod(dim2) == 30);
65 qtt_CHECK(prod(dim3) == 56);
66 qtt_CHECK(prod(dim1, 0, 3) == 24);
67 qtt_CHECK(prod(dim2, 0, 2) == 30);
68 qtt_CHECK(prod(dim3, 0, 2) == 56);
69 qtt_CHECK_NOTHROW(auto ref1 = slice(dim1, 0, 1));
70 std::vector<int64_t> concat12 = {2,3,4,5,6};
71 std::vector<int64_t> concat21 = {5,6,2,3,4} ;
72 qtt_CHECK(concat12 == concat(dim1, dim2));
73 qtt_CHECK(concat21 == concat(dim2, dim1));
74 std::vector<int64_t> concat123 = {2,3,4,5,6,7,8};
