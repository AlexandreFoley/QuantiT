407 quantity_vector permute(const int64_t *permute_begin, const int64_t *permute_end,
408 const std::vector<int64_t> repetition) const
409 {
410 quantity_vector out(size());
411 size_t p = 0;
412 for (; permute_begin != permute_end;
413 ++permute_begin) // to properly break this up would while reusing the s value would require a coroutine. or
414 // lazy algorithm (i.e. range)
415 {
416 auto rep = repetition[*permute_begin];
417 auto s = std::reduce(repetition.begin(), repetition.begin() + *permute_begin, 0);
418 for (size_t i = 0; i < rep; ++i, ++p)
419 {
420 out[p] = (*this)[s + i];
421 }
422 }
423 return out;
424 }
