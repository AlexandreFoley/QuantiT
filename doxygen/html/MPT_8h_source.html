203  // stuff about the tensors, written as covariant functions
204  S to(const torch::TensorOptions &options = {}, bool non_blocking = false, bool copy = false,
205  c10::optional<c10::MemoryFormat> memory_format = c10::nullopt) const
206  { // those function can't really exist outside the derived class if empty_copy is private, which it is for MPT, MPS
207  // and MPO.
208  S out = S::empty_copy(static_cast<S const &>(*this));
209  auto out_it = out.begin();
210  std::for_each(this->cbegin(), this->cend(),
211  [&out_it, &options, non_blocking, copy, memory_format](const auto &atensor)
212  { *(out_it++) = atensor.to(options, non_blocking, copy, memory_format); });
213  return out;
214  }
215  S to(torch::Device device, torch::ScalarType dtype, bool non_blocking = false, bool copy = false,
216  c10::optional<c10::MemoryFormat> memory_format = c10::nullopt) const
217  {
218  S out = S::empty_copy(static_cast<S const &>(*this));
219  auto out_it = out.begin();
220  std::for_each(this->cbegin(), this->cend(),
221  [&out_it, device, dtype, non_blocking, copy, memory_format](const auto &atensor)
222  { *(out_it++) = atensor.to(device, dtype, non_blocking, copy, memory_format); });
223  return out;
224  }
225  S to(torch::ScalarType dtype, bool non_blocking = false, bool copy = false,

228  S out = S::empty_copy(static_cast<S const &>(*this));

238  S out = S::empty_copy(static_cast<S const &>(*this));

245  S to(const tensor &other, bool non_blocking = false, bool copy = false,
246  c10::optional<c10::MemoryFormat> memory_format = c10::nullopt) const
247  {
248  S out = S::empty_copy(static_cast<S const &>(*this));
249  auto out_it = out.begin();
250  std::for_each(this->cbegin(), this->cend(),
251  [&out_it, &other, non_blocking, copy, memory_format](const auto &atensor)
252  { *(out_it++) = atensor.to(other, non_blocking, copy, memory_format); });
253  return out;
254  }
255  // inplace version of to, will resolve to any equivalent out-of-place equivalent.
256  template <class... Args>
257  S &to_(Args &&...args)
258  {
259  // torch::Tensor is really a pointer type, so copying it is pretty cheap: only the meta data get copied and
260  // shuffled around
261  // perhaps this could be done more efficiently, but the gains would almost certainly be incommensurate to the
262  // efforts.
263  auto other = this->to(std::forward<Args>(args)...);
264  swap(other);
265  return static_cast<S &>(*this);
266  }
