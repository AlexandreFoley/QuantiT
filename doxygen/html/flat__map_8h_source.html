347  // specialisation for the case of a inserting an ordered array, we can take some small shortcut in that case.
348  // should be templated on the iterator such that it accept iterator to ordered array with a value_type convertible
349  // to this value_type.
350  template <class Collision, class NoCollision>
351  void insert(const_iterator first, const_iterator last, Collision &&collision, NoCollision &&nocollision)
352  {
353  auto look = begin();
354  {
355  auto n1 = std::distance(first, last); // n1 is the number of element to be copied.
356  // in first approxiamtion, it's the number of element in the input
357  // detect collisions between content and the input iterators.
358  for (auto i = first; i != last; ++i)
359  {
360  look = std::lower_bound(look, end(), *i, comp);
361  if (look == end())
362  break;
363  if (!comp(*i, *look))
364  {
365  --n1; // correct it by counting all the collisions.
366  }
367  }
368  content.resize(content.size() +
369  n1); // potentially invalidates iterators. could be more time efficient if we need to
370  // reallocate, at the cost of holding the previous memory space abit longer
371  // Copy the stuff from the input that has a larger key than the largest key in this.
372  look = end() - n1; // the pointer past the end of the original, to be moved, elements
373  } // pop n1 from the stack
374  auto move_to_end = end(); // for copy backward
375  // potential optimization: if move_to_end==look at this point, all new elements are colliding.
376  while (look != begin() and last != first)
377  {
378  decltype(first - last) found_collision;
379  // part A: copy element in [first,last[ larger than *(look-1) at move_to_end.
380  {
381  auto la = std::lower_bound(first, last, *(look - 1), comp);
382  found_collision = la != last and !comp(*(look - 1), *la); // checked wether the lowest on the range to copy is colliding
383  std::copy_backward(la + found_collision, last, move_to_end); //we don't copy the colliding element
384  auto old_move_to_end = move_to_end;
385  auto move_shift = last - (la + found_collision);
386  move_to_end -= move_shift ;
387  std::for_each(move_to_end, old_move_to_end, [nocollision](auto &&x) { nocollision(std::get<1>(x)); });
388  if (found_collision)
389  {
390  --move_to_end;
391  --look;
392  if(move_to_end != look) *move_to_end = std::move( *(look));
393  collision(move_to_end->second,la->second);//can be called on a parallel thread, the tensor will not be touched again.
394  }
395  last = la;
396  } // pop la from the stack
397  // part B: move the elements in [begin,look[ that are greater than *(last-1) to move_to_end
398  if (last == first)
399  break; // we've copied everything, there's nothing to move. //ignore collision on the first elements...
400  {
401  auto lb = std::lower_bound(begin(), look, *(last - 1), comp);
402  if (move_to_end != look)
403  std::move_backward(lb, look,
404  move_to_end); // we don't have to do anything here if move_to_end == look. plus
405  // move_backward behavior is undefined in that case (depends on the implementation of the element's move).
406  // if (found_collision)
407  // collision((move_to_end - 1)->second,
408  // last->second); // if we move the collision treatment here, we allow parallelism.
409  move_to_end -= look - lb;
410  look = lb;
411  } // pop lb from the stack
412  found_collision = last != first and !comp(*(last - 1), *move_to_end);
413  if (found_collision)
414  collision(move_to_end->second,
415  (last - 1)->second); // if we move collision treatment here, to allow parallelism
416  last -= found_collision;
417  }
418  auto lastmfirst = last-first;
419  auto movembegin = move_to_end - begin();
420  assert(
421  lastmfirst == 0 or
422  movembegin == lastmfirst); // we've copied everything, or we still have some stuff to copy and the room necessary
423  std::copy_backward(first, last, move_to_end);
424  std::for_each(begin(), move_to_end, [nocollision](auto &&x) { nocollision(std::get<1>(x)); });
425  }
