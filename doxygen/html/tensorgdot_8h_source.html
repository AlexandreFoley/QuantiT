88{
89 using namespace torch;
90 Tensor A = torch::rand({5, 7, 6});
91 Tensor B = torch::rand({5, 6, 7});
92 Tensor out = torch::rand({5, 5});
93 Tensor out_copy = clone(out);
94 std::vector<int64_t> dims1 = {1, 2};
95 std::vector<int64_t> dims2 = {2, 1};
96 out += tensordot(A, B, dims1, dims2);
97 qtt_SUBCASE("tensorgdot")

99 auto new_out = tensorgdot(out_copy, A, B, dims1, dims2);
100 qtt_CHECK(allclose(new_out, out));
101 }
102 qtt_SUBCASE("tensorgdot")

104 tensorgdot_(out_copy, A, B, dims1, dims2);
105 qtt_CHECK(allclose(out_copy, out));
106 }
107 qtt_SUBCASE("tensorgdot_out")
108 {
109 Tensor out_copy2 = clone(out_copy);
110 tensorgdot_out(out_copy, out_copy2, A, B, dims1, dims2);
111 qtt_CHECK(allclose(out_copy, out));
112 }
113 qtt_SUBCASE("tensorgdot_out v2")
114 {
115 //this usage is equivalent to a call to tensordot_
116 tensorgdot_out(out_copy, out_copy, A, B, dims1, dims2);

118 }
119 qtt_SUBCASE("tensorgdot no permutation")
